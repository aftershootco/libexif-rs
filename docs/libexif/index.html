The exif crate provides a safe wrapper around the libexif C library. It provides the ability to read EXIF data from image files. The entry point for inspecting a file’s EXIF data is Data::open(). EXIF data can be inspected by iterating over the data’s contents and entries:

fn dump_exif<P: AsRef<Path>>(file_name: P) -> Result<(), libexif::ExifError> {
    let data = libexif::Data::open("image.jpg")?;

    for content in data.contents() {
        println!("[{:=>32}{:=>46}]", format!(" {:?} ", content.ifd()), "");

        for entry in content.entries() {
            println!("  {:<30} = {}",
                     entry.tag().title(content.ifd()),
                     entry.text_value()?);
        }
    }

    Ok(())
}
